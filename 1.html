// JavaScript để thực hiện slider const slider = document.querySelector('.slider'); const slides = document.querySelectorAll('.slide'); let index = 0; // Hàm để di chuyển slider function slide() { index++; if (index === slides.length) { index = 0; } const size = slides[0].clientWidth; slider.querySelector('.slides').style.transform = `translateX(${-size * index}px)`; } // Tự động chuyển slide sau một khoảng thời gian setInterval(slide, 5000); // Thay đổi thời gian ở đây nếu bạn muốn
